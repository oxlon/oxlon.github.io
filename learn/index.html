02# · learning outcome: ship a working weekly forecast

04from oxlon.data import ons, bank_of_england
05from oxlon.models import MIDAS, GradientBoost

07df = ons.pull("gdp_monthly", since="2010-01")
08macros = bank_of_england.indicators("weekly")
09
10model = MIDAS(target="gdp_qoq", high_freq=macros)
11model.fit(df, until="2024-Q4")
